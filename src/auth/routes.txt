def layout_login():

    children=[

        dcc.Input(

def register_callbacks(app):

    State("login-username", "value"),
    State("login-password", "value")
)
def login(n_clicks, username, password):
    if not n_clicks:
        return ""

    user = authenticate(username, password)
    if not user:
        return "Credenciais inválidas."
